```yaml
name: E2E Tests

on:
  # E2E-05: Schedule - daily at 2am UTC
  schedule:
    - cron: '0 2 * * *'
  # E2E-06: Push to main
  push:
    branches:
      - main
  # E2E-06: Pull requests (non-draft filtered in job condition)
  pull_request:
    types: [opened, synchronize, reopened]

# Cancel in-progress runs on same ref for efficiency
concurrency:
  group: e2e-${{ github.ref }}
  cancel-in-progress: true

jobs:
  e2e:
    # E2E-06: Skip draft PRs
    if: github.event.pull_request.draft == false || github.event_name != 'pull_request'
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 1

      # E2E-02: Docker layer caching via GHA cache backend
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Cache Docker layers
        uses: actions/cache@v4
        with:
          path: /tmp/.buildx-cache
          key: ${{ runner.os }}-buildx-${{ github.sha }}
          restore-keys: |
            ${{ runner.os }}-buildx-

      # Build Docker image with layer caching
      - name: Build Docker image
        uses: docker/build-push-action@v5
        with:
          context: .
          file: ./Dockerfile.e2e
          push: false
          load: true
          tags: openpaul-e2e:latest
          cache-from: type=local,src=/tmp/.buildx-cache
          cache-to: type=local,dest=/tmp/.buildx-cache-new,mode=max

      # Move cache to avoid cache growing indefinitely
      - name: Move Docker cache
        run: |
          rm -rf /tmp/.buildx-cache
          mv /tmp/.buildx-cache-new /tmp/.buildx-cache

      # E2E-03: Run E2E tests with retry (max 2 attempts)
      - name: Run E2E tests (attempt 1)
        id: e2e_attempt1
        continue-on-error: true
        run: |
          docker run --rm \
            -v ${{ github.workspace }}:/app \
            openpaul-e2e:latest \
            npm run test:e2e

      - name: Run E2E tests (retry on failure)
        id: e2e_attempt2
        if: steps.e2e_attempt1.outcome == 'failure'
        run: |
          echo "First attempt failed, retrying..."
          docker run --rm \
            -v ${{ github.workspace }}:/app \
            openpaul-e2e:latest \
            npm run test:e2e

      # E2E-04: Upload failure artifacts
      - name: Collect failure artifacts
        if: failure() || steps.e2e_attempt1.outcome == 'failure'
        run: |
          mkdir -p artifacts
          # Collect Docker logs
          docker logs $(docker ps -lq) > artifacts/docker-logs.txt 2>&1 || echo "No container logs available"
          # Collect .openpaul state if exists
          if [ -d ".openpaul" ]; then
            cp -r .openpaul artifacts/openpaul-state
          fi
          # Collect coverage if exists
          if [ -d "coverage" ]; then
            cp -r coverage artifacts/coverage
          fi

      - name: Upload failure artifacts
        if: failure() || steps.e2e_attempt1.outcome == 'failure'
        uses: actions/upload-artifact@v4
        with:
          name: e2e-failure-artifacts
          path: artifacts/
          retention-days: 7

      # Final status check
      - name: Check E2E test results
        if: steps.e2e_attempt1.outcome == 'failure' && steps.e2e_attempt2.outcome == 'failure'
        run: |
          echo "E2E tests failed after retry"
          exit 1
```

Key design decisions:
- Uses same action versions as test.yml (@v4 conventions)
- Draft PR filter matches test.yml pattern
- Concurrency group prevents redundant E2E runs
- Docker layer caching via actions/cache for faster subsequent builds
- Retry logic: continue-on-error on first attempt, second attempt only runs on failure
- Failure artifacts include Docker logs, .openpaul state, and coverage
- 7-day retention matches test.yml artifact retention
