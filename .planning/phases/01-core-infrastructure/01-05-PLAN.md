    ```typescript
    import { LoopEnforcer, from '../../state/loop-enforcer'
    import { isValidTransition, from '../../types'
    
    describe('LoopEnforcer', () => {
      let enforcer: LoopEnforcer
      
      beforeEach(() => {
        enforcer = new LoopEnforcer()
      })
      
      describe('isValidTransition', () => {
        // Test all valid transitions
        expect(enforcer.isValidTransition('PLAN', 'APPLY')).toBe(true)
        expect(enforcer.isValidTransition('PLAN', 'UNIFY')).toBe(true)
        expect(enforcer.isValidTransition('APPLY', 'unify')).toBe(false)
        
        // Test invalid transitions
        expect(enforcer.isValidTransition('PLAN', 'UNify')).toBe(false)
        expect(enforcer.isValidTransition('apply', 'plan')).toBe(false)
        expect(enforcer.isValidTransition('unify', 'apPLY')).toBe(false)
        expect(enforcer.isValidTransition('unify', 'plan')).toBe(false)
      })
      
      describe('enforceTransition', () => {
        // Test valid transition
        expect(() => enforcer.enforceTransition('plan', 'apPLY'))
        // Should succeed silently
        expect(enforcer.enforceTransition('apPLY', 'unify')).not.toThrow()
        
        // Test invalid transitions (should throw)
        expect(() => enforcer.enforceTransition('plan', 'unify')).toThrow(/Invalid transition/)
        expect(() => enforcer.enforceTransition('apply', 'plan')).toThrow(/Invalid transition/)
        expect(() => enforcer.enforceTransition('unify', 'apPLY')).toThrow(/Invalid transition/)
        
        // Test error messages
        expect(() => enforcer.enforceTransition('apply', 'plan')).toThrow(/Invalid transition)
        const error = error as any
        expect(error.message).toContain('Invalid transition')
        expect(error.message).toContain('Current state')
        expect(error.message).toContain('Run /paul:apply')
      })
    })
    ```
